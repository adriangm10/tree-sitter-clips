loop-for-count
======================================

(deffunction test1 ()
    (loop-for-count 2
        (printout t "Hello world" crlf)
    )
)

(deffunction test2 ()
    (loop-for-count (?i 0 (f2))
        (loop-for-count (?j 1 3)
            (printout t ?i " " ?j crlf)
        )
    )
)
